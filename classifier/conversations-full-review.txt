# Define a function to extract images from doc and docx files
def extract_images_from_docx(doc_file):
    img_folder = doc_file[:-4] + '_images'
    if not os.path.exists(img_folder):
        os.mkdir(img_folder)
    with zipfile.ZipFile(doc_file, 'r') as z_file:
        for name in z_file.namelist():
            if name.startswith('word/media/'):
                z_file.extract(name, img_folder)
                
# Define a function to extract images from a folder containing doc and docx files
def extract_images_from_folder(folder_path):
    for file_name in os.listdir(folder_path):
        if file_name.endswith('.docx'):
            doc_file = os.path.join(folder_path, file_name)
            extract_images_from_docx(doc_file)
        elif file_name.endswith('.doc'):
            doc_file = os.path.join(folder_path, file_name)
            docx_file = os.path.join(folder_path, f'{file_name}x')
            try:
                shutil.copy(doc_file, docx_file)
                extract_images_from_docx(docx_file)
                os.remove(docx_file)
            except:
                print(f'Error occurred while processing {doc_file}')
        else:
            print(f'Skipping file {file_name}')
            
# Define a function to convert images to JPG format
def convert_images_to_jpg(img_folder):
    for name in os.listdir(img_folder):
        if name.endswith(('png', 'jpeg', 'gif', 'tiff', 'bmp')):
            img_path = os.path.join(img_folder, name)
            with Image.open(img_path) as img:
                img.save(os.path.join(img_folder, f'{name[:-4]}.jpg'), 'JPEG')

# Define a function to add images to a Word document
def add_images_to_docx(doc_file, img_folder):
    word_doc = Document(doc_file)
    for para in word_doc.paragraphs:
        for run in para.runs:
            if run.alignment == WD_ALIGN_PARAGRAPH.CENTER:
                img_name = run._r.graphic.graphicData.pic.nvPicPr.cNvPr.name
                img_path = os.path.join(img_folder, img_name)
                if os.path.exists(img_path):
                    para.insert_paragraph_before()
                    para = para.previous_paragraph
                    para.add_run().add_picture(img_path, width=Image.open(img_path).width/1.6)
